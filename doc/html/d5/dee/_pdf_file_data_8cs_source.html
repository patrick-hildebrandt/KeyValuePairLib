142 private string GetPdfPageString(PdfPageBase pdfPage)
143 {
144 // ! Determine the size of the PDF page
145 int pdfWidth = (int)pdfPage.ActualSize.Width;
146 int pdfHeight = (int)pdfPage.ActualSize.Height;
147 // ! Instantiation of PdfTextExtractor class for extraction
148 PdfTextExtractor pdfTextExtractor = new(pdfPage);

150 PdfTextExtractOptions extractionOptions = new();
151 // ! Define the area for extraction
152 extractionOptions.ExtractArea = new RectangleF(_config.CutLeft, _config.CutTop, pdfWidth -
153 _config.CutRight, pdfHeight - _config.CutBottom);
154 // ? Effect unknown / default = false
155 //extractionOptions.IsExtractAllText = false;

158 // ! Encode string to UTF8
159 pdfPageStr = Encoding.UTF8.GetString(Encoding.Convert(Encoding.Default, Encoding.UTF8,
160 Encoding.Default.GetBytes(pdfPageStr)));
161 // ! Mask characters used in regex
162 pdfPageStr = _regexUtils.MaskRegexChars(pdfPageStr);
163 // ! Remove Spire.PDF watermark

189 private static string GetPdfDocEmptyRowsRemoved(string pdfDocStr)
190 {
191 // ! Return string
192 string pdfDocEmptyRowsRemoved = pdfDocStr;
193 // ! STRINGSTART-WHITESPACE zero or more-RETURN optional-NEWLINE OR
194 // ! RETURN optional-NEWLINE-WHITESPACE zero or more-STRINGEND
195 string startEndEmptyRows = @"^\s*\r?\n|\r?\n\s*$";
196 // ! Remove empty rows at start and end
197 pdfDocEmptyRowsRemoved = Regex.Replace(pdfDocEmptyRowsRemoved, startEndEmptyRows, "");
198 // ! (RETURN optional-NEWLINE){QUANTIFIER 2 or more}
199 string doubleRowBreaks = @"(\r?\n){2,}";
200 // ! Remove double row breaks
201 pdfDocEmptyRowsRemoved = Regex.Replace(pdfDocEmptyRowsRemoved, doubleRowBreaks, "\r\n");
202 // ! (RETURN optional-NEWLINE-WHITESPACE zero or more-RETURN optional-NEWLINE)
203 string innerEmptyRows = @"(\r?\n\s*\r?\n)";
204 // ! Remove remaining inner empty rows
205 pdfDocEmptyRowsRemoved = Regex.Replace(pdfDocEmptyRowsRemoved, innerEmptyRows, "\r\n");
206 // ! Return
207 return pdfDocEmptyRowsRemoved;
208 }
215 private string GetPdfDocFieldsDelimited(string pdfDocEmptyRowsRemoved)
216 {
217 // ! Return string
218 string pdfDocFieldsDelimited = pdfDocEmptyRowsRemoved;
219 // ! Empty "rooms" are replaced with "\u0000\t\t\u0000"
220 pdfDocFieldsDelimited = Regex.Replace(pdfDocFieldsDelimited,
221 // ! (NON-WHITESPACE)(SPACE{QUANTIFIER MaxWhiteSpaces}), backreference capture group 1 => \u0000\t\t\u0000
222 @"(\S)( {" + _config.MaxWhiteSpaces + @"})", "$1\u0000\t\t\u0000");
223 pdfDocFieldsDelimited = Regex.Replace(pdfDocFieldsDelimited, @"^", "\u0000\t\t\u0000");
224 pdfDocFieldsDelimited = Regex.Replace(pdfDocFieldsDelimited, @"$", "\u0000\t\t\u0000");
225 pdfDocFieldsDelimited = Regex.Replace(pdfDocFieldsDelimited, @"\r\n",
226 "\u0000\t\t\u0000\r\n\u0000\t\t\u0000");
227 while (pdfDocFieldsDelimited.Contains("\u0000\t\t\u0000 ")) pdfDocFieldsDelimited = pdfDocFieldsDelimited
228 .Replace("\u0000\t\t\u0000 ", "\u0000\t\t\u0000");
229 while (pdfDocFieldsDelimited.Contains(" \u0000\t\t\u0000")) pdfDocFieldsDelimited = pdfDocFieldsDelimited
230 .Replace(" \u0000\t\t\u0000", "\u0000\t\t\u0000");
231 pdfDocFieldsDelimited = pdfDocFieldsDelimited.Replace("\u0000\t\t\u0000\u0000\t\t\u0000",
232 "\u0000\t\t\u0000");
233 // ! Return
234 return pdfDocFieldsDelimited;
235 }
242 private Dictionary<int, string> GetRows(string pdfDocEmptyRowsRemoved)
243 {
244 // ! Return Dictionary
245 Dictionary<int, string> rows = new();
246 // ! Split rows to List
247 List<string> splitRows = pdfDocEmptyRowsRemoved.Split("\r\n").ToList();
248 // ! Remove row breaks
249 for (int i = 0; i < splitRows.Count; i++) splitRows[i] = splitRows[i].Replace("\r\n", "");
250 // ! Add rows to Dictionary with an index
251 for (int i = 0; i < splitRows.Count; i++)
252 {
253 // ! Buffer corresponding to RangeBuffer is inserted at STRINGSTART
254 rows[i] = Regex.Replace(splitRows[i], @"^", new string(' ', _config.RangeBuffer));
255 }
256 return rows;
257 }
264 private static Dictionary<int, List<string>> GetFields(string pdfDocFieldsDelimited)
265 {
266 // ! Return Dictionary
267 Dictionary<int, List<string>> fields = new();
268 // ! Split rows to List
269 List<string> rows = pdfDocFieldsDelimited.Split("\r\n").ToList();
270 // ! Remove row breaks
271 for (int i = 0; i < rows.Count; i++) rows[i] = rows[i].Replace("\r\n", "");
272 // ! Provide rows with an index and split fields
273 for (int i = 0; i < rows.Count; i++)
274 {
275 if (!fields.ContainsKey(i)) fields.Add(i, new List<string>());
276 fields[i] = rows[i].Split("\u0000\t\t\u0000").ToList();
277 fields[i].RemoveAll(string.IsNullOrEmpty);
278 }
279 return fields;
280 }
288 private static Dictionary<int, List<(string, int, int)>> GetIndexedFields(Dictionary<int, string> rows,
289 Dictionary<int, List<string>> fields)
290 {
291 // ! Return Dictionary
292 Dictionary<int, List<(string, int, int)>> indexedFields = new();
293 // ! Iterate through rows
294 for (int i = 0; i < fields.Count; i++)
295 {
296 // ! Needed to move search area
297 int nextIndex = 0;
298 // ! Iterate through fields
299 foreach (string field in fields[i])
300 {
301 // ! Create key if not exists
302 if (!indexedFields.ContainsKey(i)) indexedFields[i] = new List<(string, int, int)>();
303 // ! Field (with start and end index) is stored or searched for after index of last field
304 indexedFields[i].Add((field, rows[i].IndexOf(field, nextIndex), rows[i].IndexOf(field, nextIndex) +
305 field.Length));
306 // ! Move search area
307 nextIndex = rows[i].IndexOf(field) + field.Length + 1;
308 }
309 }
310 return indexedFields;
311 }
318 private static DataTable GetFieldTable(Dictionary<int, List<(string, int, int)>> indexedFields)
319 {
320 // ! Return DataTable
321 DataTable dataTable = new();
322 // ! Rowindex
323 dataTable.Columns.Add("Row", typeof(int));
324 // ! Startindex
325 dataTable.Columns.Add("Column", typeof(int));
326 // ! Value
327 dataTable.Columns.Add("Value", typeof(string));
328 // ! Endindex
329 dataTable.Columns.Add("End", typeof(int));
330 // ! Iterate through rows
331 for (int i = 0; i < indexedFields.Count; i++)
332 {
333 // ! Iterate through columns
334 foreach ((string, int, int) entry in indexedFields[i]) dataTable.Rows.Add(i, entry.Item2, entry.Item1,
335 entry.Item3);
336 }
337 // ! Convert DataTable to DataView
338 DataView dataView = dataTable.DefaultView;
339 // ! Sort in DataView
340 dataView.Sort = "Row ASC, Column ASC";
341 // ! Convert back to DataTable
342 return dataView.ToTable();
